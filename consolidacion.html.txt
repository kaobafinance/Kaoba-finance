<script>
let deudaCount = 0;
let chartInstance = null;

function addDeuda() {
deudaCount++;
const container = document.getElementById('deudas-container');
const div = document.createElement('div');
div.className = 'deuda-block';
div.id = `deuda-${deudaCount}`;
div.innerHTML = `
<button class="remove-btn" onclick="removeDeuda(${deudaCount})">×</button>
<label>Tipo de deuda:
<select id="tipo-${deudaCount}">
<option value="personal">Préstamo personal</option>
<option value="tarjeta">Tarjeta de crédito</option>
<option value="linea">Línea de crédito</option>
</select>
</label>
<label>Capital pendiente (€):
<input type="number" id="capital-${deudaCount}" value="0" min="0" onfocus="clearZero(this)">
</label>
<label>Tipo de interés anual (TIN %):
<input type="number" id="interes-${deudaCount}" value="0" step="0.01" min="0" onfocus="clearZero(this)">
</label>
<label>Plazo restante (meses):
<input type="number" id="plazo-${deudaCount}" value="0" min="1" onfocus="clearZero(this)">
</label>`;
container.appendChild(div);
}

function clearZero(input){ if(input.value==='0') input.value=''; }
function removeDeuda(id){ const div=document.getElementById(`deuda-${id}`); if(div) div.remove(); }

function validarInputs(deudas,plazoNuevo,interesNuevo){
if(deudas.length===0){ alert("Agrega al menos una deuda"); return false;}
for(let d of deudas){ if(d.capital<=0 || d.interes<0 || d.plazo<=0){ alert("Todos los campos de deuda deben ser positivos"); return false;} }
if(plazoNuevo<=0 || interesNuevo<0){ alert("Datos del préstamo nuevo inválidos"); return false;}
return true;
}

function calcularCuota(capital,interesAnual,plazoMeses){ const i=interesAnual/100/12; return (capital*i*Math.pow(1+i,plazoMeses))/(Math.pow(1+i,plazoMeses)-1); }

function calcularConsolidacion(){
const deudas=[];
for(let i=1;i<=deudaCount;i++){ const div=document.getElementById(`deuda-${i}`); if(!div) continue; const capital=parseFloat(document.getElementById(`capital-${i}`).value)||0; const interes=parseFloat(document.getElementById(`interes-${i}`).value)||0; const plazo=parseInt(document.getElementById(`plazo-${i}`).value)||0; deudas.push({capital,interes,plazo}); }
const plazoNuevo=parseInt(document.getElementById('plazoNuevo').value);
const interesNuevo=parseFloat(document.getElementById('interesNuevo').value);
const comisiones=parseFloat(document.getElementById('comisionesNuevo').value)||0;
const pagoExtra=parseFloat(document.getElementById('pagoExtra').value)||0;
if(!validarInputs(deudas,plazoNuevo,interesNuevo)) return;

const totalCapital=deudas.reduce((sum,d)=>sum+d.capital,0);
const cuotaAntes=deudas.reduce((sum,d)=>sum+calcularCuota(d.capital,d.interes,d.plazo),0);
const cuotaDespues=calcularCuota(totalCapital+comisiones,interesNuevo,plazoNuevo)+pagoExtra;
const interesesAntes=deudas.reduce((sum,d)=>sum+(calcularCuota(d.capital,d.interes,d.plazo)*d.plazo-d.capital),0);
const interesesDespues=cuotaDespues*plazoNuevo-totalCapital-comisiones;

document.getElementById('cuotaAntes').innerText=cuotaAntes.toFixed(2)+' €';
document.getElementById('cuotaDespues').innerText=cuotaDespues.toFixed(2)+' €';
document.getElementById('capitalAntes').innerText=totalCapital.toFixed(2)+' €';
document.getElementById('capitalDespues').innerText=totalCapital.toFixed(2)+' €';
document.getElementById('interesesAntes').innerText=interesesAntes.toFixed(2)+' €';
document.getElementById('interesesDespues').innerText=interesesDespues.toFixed(2)+' €';
document.getElementById('ahorroMensual').innerText=(cuotaAntes-cuotaDespues).toFixed(2)+' €';
document.getElementById('ahorroTotal').innerText=(interesesAntes-interesesDespues).toFixed(2)+' €';

document.getElementById('resultados').style.display='block';

const ctx=document.getElementById('graficoCuotas').getContext('2d');
if(chartInstance) chartInstance.destroy();
chartInstance=new Chart(ctx,{type:'bar',data:{labels:['Cuota Mensual','Intereses Totales'],datasets:[{label:'Antes',data:[cuotaAntes,interesesAntes],backgroundColor:'#0d6efd'},{label:'Después',data:[cuotaDespues,interesesDespues],backgroundColor:'#198754'}]},options:{responsive:true}});
}

// PDF + envío correo con validación RGPD
function descargarPDFCorreo(){
const email=document.getElementById('emailUsuario').value.trim();
const acepta=document.getElementById('aceptaDatos').checked;

if(!email || !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)){
alert("Introduce un correo válido");
return;
}

if(!acepta){
alert("Debes aceptar la política de privacidad para enviar el PDF");
return;
}

const { jsPDF }=window.jspdf;
const doc=new jsPDF();
doc.setFontSize(16);
doc.text("Simulación Consolidación de Deudas - Kaoba Finance",20,20);
const tabla=[["Concepto","Antes","Después"],["Cuota Mensual",document.getElementById('cuotaAntes').innerText,document.getElementById('cuotaDespues').innerText],["Capital Total",document.getElementById('capitalAntes').innerText,document.getElementById('capitalDespues').innerText],["Intereses Totales",document.getElementById('interesesAntes').innerText,document.getElementById('interesesDespues').innerText],["Ahorro Mensual","-",document.getElementById('ahorroMensual').innerText],["Ahorro Total","-",document.getElementById('ahorroTotal').innerText]];
doc.autoTable({head:[tabla[0]],body:tabla.slice(1),startY:30});
const pdfBlob=doc.output('blob');

// Enviar a servidor (endpoint) para correo
const formData=new FormData();
formData.append('pdf',pdfBlob,'Simulacion_Consolidacion.pdf');
formData.append('email',email);

fetch('/enviar-pdf-correo', { method:'POST', body:formData })
.then(resp=>resp.json())
.then(data=>{ alert(data.message || 'PDF enviado al correo'); })
.catch(err=>{ console.error(err); alert('Error enviando PDF'); });
}

function descargarCSV(){
const rows=[["Concepto","Antes","Después"],["Cuota Mensual",document.getElementById('cuotaAntes').innerText,document.getElementById('cuotaDespues').innerText],["Capital Total",document.getElementById('capitalAntes').innerText,document.getElementById('capitalDespues').innerText],["Intereses Totales",document.getElementById('interesesAntes').innerText,document.getElementById('interesesDespues').innerText],["Ahorro Mensual","-",document.getElementById('ahorroMensual').innerText],["Ahorro Total","-",document.getElementById('ahorroTotal').innerText]];
let csvContent="data:text/csv;charset=utf-8,"+rows.map(e=>e.join(",")).join("\n");
const encodedUri=encodeURI(csvContent);
const link=document.createElement("a");
link.setAttribute("href",encodedUri);
link.setAttribute("download","Simulacion_Consolidacion.csv");
document.body.appendChild(link); link.click(); document.body.removeChild(link);
}
